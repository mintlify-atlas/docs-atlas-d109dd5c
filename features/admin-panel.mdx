### Extraction Logic

From `admin.html:725-747`:

```javascript
function extractVideoId(raw, fuente) {
    const value = (raw || "").trim();
    if (!value) return "";
    if (!value.includes("http")) return value;

    try {
        const url = new URL(value);
        if (fuente === "youtube") {
            if (url.hostname.includes("youtu.be")) return url.pathname.replace("/", "");
            const v = url.searchParams.get("v");
            if (v) return v;
            const segments = url.pathname.split("/").filter(Boolean);
            return segments[segments.length - 1] || value;
        }
        if (fuente === "drive") {
            const match = value.match(/\/d\/([^/]+)/);
            if (match && match[1]) return match[1];
            const idParam = url.searchParams.get("id");
            if (idParam) return idParam;
        }
    } catch {}
    return value;
}
```

Supported URL formats:

**YouTube:**
- `https://www.youtube.com/watch?v=VIDEO_ID`
- `https://youtu.be/VIDEO_ID`
- `https://www.youtube.com/embed/VIDEO_ID`

**Google Drive:**
- `https://drive.google.com/file/d/FILE_ID/view`
- `https://drive.google.com/open?id=FILE_ID`

<Tip>
  The system also auto-extracts IDs when you blur the videoId field, saving you from clicking the "Extract" button manually.
</Tip>

## Save Operations

### Create New Song

From `admin.html:749-792`:

```javascript
async function guardarTema() {
    const editId = refs.editId.value;
    const data = collectData();
    data.videoId = extractVideoId(data.videoId, data.fuente);

    const err = validateData(data);
    if (err) {
        showStatus(err, false);
        return;
    }

    refs.btnGuardar.disabled = true;
    refs.btnGuardar.textContent = editId ? "Actualizando..." : "Guardando...";

    try {
        if (editId) {
            await updateDoc(doc(db, "temas", editId), data);
            const idx = allTemas.findIndex(t => t._id === editId);
            if (idx > -1) allTemas[idx] = { ...allTemas[idx], ...data };
            showStatus("✓ Tema actualizado.", true);
        } else {
            const payload = { ...data, createdAt: serverTimestamp() };
            const ref = await addDoc(collection(db, "temas"), payload);
            allTemas.push({ _id: ref.id, ...data, createdAtMs: Date.now() });
            showStatus("✓ Tema añadido.", true);
        }

        limpiar();
        updateStats();
        applyFiltersAndRender();
    } catch (e) {
        console.error(e);
        showStatus("Error: " + e.message, false);
    } finally {
        refs.btnGuardar.disabled = false;
        // ... restore button text
    }
}
```

<Note>
  New songs automatically receive a `createdAt` timestamp using Firestore's `serverTimestamp()` for accurate sorting.
</Note>

function renderTabla(lista) {
    refs.tableCount.textContent = `${lista.length} tema${lista.length !== 1 ? "s" : ""}`;
    if (!lista.length) {
        refs.tbody.innerHTML = `<tr class="empty-row"><td colspan="6">Sin resultados</td></tr>`;

    }

    refs.tbody.innerHTML = lista.map(t => `
        <tr>
            <td>${t.titulo || "—"}</td>
            <td>${t.artista || "—"}</td>

            <td><span class="cat-badge">${t.categoria || "—"}</span></td>

                    <button class="btn-dup-row" onclick="duplicar('${t._id}')">Duplicar</button>
                    <button class="btn-copy-row" onclick="copiarLink('${t._id}','${t.categoria || "huayno"}')">Link</button>
                    <button class="btn-del-row" onclick="borrar('${t._id}')">Eliminar</button>
                </div>
            </td>
        </tr>`).join("");

    refs.lastUpdate.textContent = `Actualizado ${fmtNow()}`;
}

### Filter Implementation

From `admin.html:628-656`:

```javascript
function applyFiltersAndRender() {
    const q = normalizar(refs.search.value);
    const cat = refs.filterCategoria.value;
    const src = refs.filterFuente.value;
    const sortMode = refs.sort.value;

    filteredTemas = allTemas.filter(t => {
        const matchesText = normalizar(t.titulo).includes(q)
            || normalizar(t.artista).includes(q)
            || normalizar(t.tonalidad).includes(q);
        const matchesCat = cat === "all" || (t.categoria || "") === cat;
        const matchesSrc = src === "all" || (t.fuente || "") === src;
        return matchesText && matchesCat && matchesSrc;
    });

    filteredTemas.sort((a, b) => {
        const ta = (a.titulo || "").toLowerCase();
        const tb = (b.titulo || "").toLowerCase();
        const aa = (a.artista || "").toLowerCase();
        const ab = (b.artista || "").toLowerCase();

        if (sortMode === "titulo-asc") return ta.localeCompare(tb);
        if (sortMode === "titulo-desc") return tb.localeCompare(ta);
        if (sortMode === "artista-asc") return aa.localeCompare(ab);
        return (b.createdAtMs || 0) - (a.createdAtMs || 0);
    });

    renderTabla(filteredTemas);
}
```

### Duplicate

From `admin.html:815-838`:

```javascript
window.duplicar = async (id) => {
    const t = allTemas.find(x => x._id === id);
    if (!t) return;
    const copia = {
        titulo: `${t.titulo || "Tema"} (copia)`,
        artista: t.artista || "",
        tonalidad: t.tonalidad || "",
        videoId: t.videoId || "",
        inicio: t.inicio || 0,
        categoria: t.categoria || "huayno",
        fuente: t.fuente || "youtube",
        createdAt: serverTimestamp()
    };

    try {
        const ref = await addDoc(collection(db, "temas"), copia);
        allTemas.unshift({ _id: ref.id, ...copia, createdAtMs: Date.now() });
        updateStats();
        applyFiltersAndRender();
        showStatus("✓ Tema duplicado.", true);
    } catch (e) {
        showStatus("Error al duplicar: " + e.message, false);
    }
};
```

## CSV Export

Export the current filtered view or entire catalog to CSV.

From `admin.html:864-885`:

```javascript
function exportCsv() {
    const rows = filteredTemas.length ? filteredTemas : allTemas;
    if (!rows.length) {
        showStatus("No hay datos para exportar.", false);
        return;
    }

    const headers = ["titulo", "artista", "tonalidad", "categoria", "fuente", "videoId", "inicio"];
    const escapeCell = (v) => `"${String(v ?? "").replace(/"/g, '""')}"`;
    const csv = [headers.join(",")]
        .concat(rows.map(r => headers.map(h => escapeCell(r[h])).join(",")))
        .join("\n");

    const blob = new Blob([csv], { type: "text/csv;charset=utf-8;" });
    const url = URL.createObjectURL(blob);
    const a = document.createElement("a");
    a.href = url;
    a.download = `ripper-temas-${new Date().toISOString().slice(0, 10)}.csv`;
    a.click();
    URL.revokeObjectURL(url);
    showStatus("✓ CSV exportado.", true);
}
```

Exported fields:
- titulo
- artista
- tonalidad
- categoria
- fuente
- videoId
- inicio

document.addEventListener("keydown", (e) => {
    const activeTag = (document.activeElement?.tagName || "").toLowerCase();
    if (e.key === "Escape") {
        limpiar();
    }
    if (e.key === "Enter" && activeTag !== "textarea" && activeTag !== "button") {

function showStatus(msg, ok) {
    refs.status.textContent = msg;

Status messages appear for:

- ✓ Duplications
- ✓ Deletions
- ✓ ID extraction
- ✗ Validation errors
- ✗ Network errors
